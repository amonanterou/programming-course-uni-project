def run_random_walk(steps):
    """              
           .
    """
    pos = np.array([0, 0]) #  

    positions = [pos] #     
    distances = [] #      
    
    #         
    for step in range(steps):
        #    
        move = rng.choice(rand_moves)
        #           
        pos = pos + move
        #           
        positions.append(pos)
        #       
        distances.append(np.linalg.norm(pos))
    
    #     ,   ,   ,       
    return positions, max(distances), min(distances), np.mean(distances)

def show_walk(positions):
    #       
    x, y = [], []
    for v in positions: #   
        x.append(v[0]) #   
        y.append(v[1]) #   
    
    # plot
    fig, ax = plt.subplots() #       (fig)     (ax)   

    ax.plot(x, y, linewidth=2.0) #     
    
    ax.set_title(" ")

    plt.show() #   

def run_simulation(steps):
    #  100    
    results = [] #      
    sim2plot = rng.integers(100) #       100
    
    #  100 
    for i in range(100):
        
        #        positions, max(distances), min(distances), np.mean(distances)
        r = run_random_walk(steps)
        #         results
        results.append(r[1:])
        
        #      
        if i == sim2plot:
            #        
            show_walk(r[0])
    
    #   
    return results
